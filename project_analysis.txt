  rateLimiter = require('../clockwork-rate-limiter');
} catch (e) {
  // Fallback if rate limiter doesn't exist
  rateLimiter = {
    auth: (req, res, next) => next(),
    strict: (req, res, next) => next(),
    default: (req, res, next) => next()
  };
